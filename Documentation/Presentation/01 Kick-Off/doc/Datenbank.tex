\frame{\frametitle{Alle Medien}
In einer Datenbank sollen folgende Details für alle Medien gespeichert werden.

\begin{itemize}
	\item Barcode
	\item Titel
	\item Art des Items (Buch, Film oder Spiel)
	\item Genre
	\item Sprache
	\item Erscheinungsjahr
	\item Bewertung 
	\item Ort der Aufbewahrung

	\item Bemerkung
\end{itemize}
